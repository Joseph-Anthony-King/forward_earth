### fe.01 - Instantiate Earth if not done so yet
event = {
	id = fe.01
	hide_window = yes
	is_triggered_only = yes

	immediate = {

		if = {
			# If there are no Sols
			limit = {
				NOT = {
					any_system = {
						has_star_flag = sol
					}
				}
			}
			random_list = {
				90 = {
					random_system = {
						limit = { has_owner = no }
						spawn_system = { initializer = "special_init_03" }
					}
				}
				10 = {
					random_system = {
						limit = { has_owner = no }
						spawn_system = { initializer = "special_init_04" }
					}
				}
			}
		}
    }
}
